def count_shifts(s: str) -> int:    amount = 0    min_amount = 0    min_count = 0    for char in s:        if char == '(':            amount += 1        elif char == ')':            amount -= 1        if amount < min_amount:            min_amount = amount            min_count = 1        elif amount == min_amount:            min_count += 1    return min_count if amount == 0 else 0def main() -> None:    seq = input().strip()    print(count_shifts(seq))if __name__ == "__main__":    main()
