def count_brackets(expression: str) -> int:    open_count = 0    min_open_count = 0    min_open_count_occurrences = 0    for char in expression:        if char == '(':            open_count += 1        elif char == ')':            open_count -= 1        if open_count < min_open_count:            min_open_count = open_count            min_open_count_occurrences = 1        elif open_count == min_open_count:            min_open_count_occurrences += 1    if open_count == 0:        return min_open_count_occurrences    return 0pos = input().strip()print(count_brackets(pos))
