def main() -> None:    """    This is the main function    :return None:    """    n = int(input())    a = [int(input()) for _ in range(n)]    sum_mod_n = [0] * (n + 1)    f = [0] * (n + 1)    for i in range(1, n + 1):        sum_mod_n[i] = (a[i - 1] + sum_mod_n[i - 1]) % n        if f[sum_mod_n[i]] or sum_mod_n[i] == 0:            print(i - f[sum_mod_n[i]])            for j in range(f[sum_mod_n[i]] + 1, i + 1):                print(a[j - 1])            return        f[sum_mod_n[i]] = iif __name__ == "__main__":    main()
