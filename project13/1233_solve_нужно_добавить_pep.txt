def z(k):    i = 0    while k:        k //= 10        i += 1    return 10 ** (i - 1)def calc(K, M):    p = [0] * 20    for i in range(19):        p[i] = 10 ** i    n = 0    k = K    while k:        n += 1 + (k - z(k))        k //= 10    if M < n:        return 0    M -= n    if M:        i = 1        while True:            l = p[i] * (K - z(K))            if not l:                return 0            if M <= l:                return M - 1 + z(K) * p[i]            M -= l            i += 1    else:        return Kdef main():    K, M = map(int, input().split())    print(calc(K, M))if __name__ == "__main__":    main()
