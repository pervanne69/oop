def main() -> None:    """    Main function to read input and solve the problem    :return None:    """    n = int(input())    pow3 = [1]    while pow3[-1] <= (2 ** 32 - 1) // 3:        pow3.append(pow3[-1] * 3)    j = 0    while n >= (pow3[j + 1] - 1) // 2:        j += 1    for k in range(1 << (j + 1)):        total = 0        for j in range(j + 1):            if k & (1 << j):                total += pow3[j]        if total <= n:            continue        tip = total - n        for l_p in range(j, -1, -1):            if tip >= pow3[l_p]:                tip -= pow3[l_p]        if tip == 0:            print(total, total - n)            return    print(0)if __name__ == "__main__":    main()
